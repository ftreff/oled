Command:
sudo nano /etc/systemd/system/oled-sys.service                      

Copy and Paste the following:

#********[COPY BELOW THIS LINE]********
[Unit]
Description=OLED System Stats Service
After=network.target
StartLimitIntervalSec=0
[Service]
Type=simple
Restart=always
RestartSec=1
User=<USER NAME>
ExecStart=python3 /home/<USER NAME>/oled/ft_sys.py
[Install]
WantedBy=multi-user.target
#********[COPY ABOVE THIS LINE]********

change <USER NAME> to your user name in path and User=

run commands:
	sudo systemctl start oled-sys
	sudo systemctl enable oled-sys

if you everwant to "uninstall" this service use command
	sudo systemctl disable oled-sys
	
if you everwant to stop this service use command
	sudo systemctl stop oled-sys
